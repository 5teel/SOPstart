    Content hash computation (inline in page or extract to utility):
    ```typescript
    async function computeContentHash(steps: SopStep[]): Promise<string> {
      const canonical = steps.sort((a, b) => a.step_number - b.step_number)
        .map(s => `${s.step_number}:${s.text}`).join('|')
      const data = new TextEncoder().encode(canonical)
      const hashBuffer = await crypto.subtle.digest('SHA-256', data)
      return Array.from(new Uint8Array(hashBuffer)).map(b => b.toString(16).padStart(2, '0')).join('')
    }
    ```
