**Implementation strategy** — use a single plpgsql DO block that loops over inline literal arrays of seed data so the entire migration is one transaction (the `blocks_current_version_fk` is DEFERRABLE INITIALLY DEFERRED so block→version→update-current_version order works):

```sql
do $$
declare
  v_block_id uuid;
  v_version_id uuid;
begin

  -- ============================================================
  -- Hazard blocks
  -- ============================================================
  -- Block 1: <name>
  insert into public.blocks (organisation_id, kind_slug, name, category_tags, free_text_tags, created_by)
    values (null, 'hazard', '<name>', ARRAY['<tag1>','<tag2>']::text[], ARRAY[]::text[], null)
    returning id into v_block_id;
  insert into public.block_versions (block_id, version_number, content, change_note, created_by)
    values (v_block_id, 1, jsonb_build_object('kind', 'hazard', 'text', '<text>', 'severity', '<severity>'), 'Initial NZ global seed (D-Global-03)', null)
    returning id into v_version_id;
  update public.blocks set current_version_id = v_version_id where id = v_block_id;

  -- Block 2: ... (repeat for every entry in JSON)

  -- ============================================================
  -- PPE blocks
  -- ============================================================
  insert into public.blocks (organisation_id, kind_slug, name, category_tags, free_text_tags, created_by)
    values (null, 'ppe', 'Safety Glasses', ARRAY['eye-injury']::text[], ARRAY[]::text[], null)
    returning id into v_block_id;
  insert into public.block_versions (block_id, version_number, content, change_note, created_by)
    values (v_block_id, 1, jsonb_build_object('kind', 'ppe', 'items', jsonb_build_array('Safety glasses (Z87.1 or AS/NZS 1337 rated)')), 'Initial NZ global seed (D-Global-03)', null)
    returning id into v_version_id;
  update public.blocks set current_version_id = v_version_id where id = v_block_id;

  -- ... (repeat for all PPE)

  -- ============================================================
  -- Step pattern blocks
  -- ============================================================
  -- For step blocks with warning/tip, use jsonb_build_object('kind', 'step', 'text', '...', 'warning', '...', 'tip', '...')

end $$;

end $$;
```
