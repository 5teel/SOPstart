   ```tsx
   'use client'
   import { useEffect } from 'react'

   export function PaperThemeMount() {
     useEffect(() => {
       const prev = document.body.getAttribute('data-theme')
       document.body.setAttribute('data-theme', 'paper')
       return () => {
         if (prev === null) document.body.removeAttribute('data-theme')

       }
     }, [])
     return null
   }
   ```

   ```sql
     on public.walkthrough_progress for insert to authenticated

   create policy "walkthrough_progress_update_own"
     on public.walkthrough_progress for update to authenticated
   ```
