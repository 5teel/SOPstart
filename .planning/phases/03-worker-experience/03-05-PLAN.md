**1. Create `supabase/migrations/00008_sop_versioning.sql`:**
```sql
-- Add versioning columns to sops table
ALTER TABLE public.sops
ADD COLUMN superseded_by uuid REFERENCES public.sops(id) ON DELETE SET NULL,
ADD COLUMN parent_sop_id uuid REFERENCES public.sops(id) ON DELETE SET NULL;

-- parent_sop_id: all versions of the "same SOP" share the same parent_sop_id (the original SOP's id)
-- superseded_by: old version points to new version (null for current version)

CREATE INDEX idx_sops_parent ON public.sops (parent_sop_id);
CREATE INDEX idx_sops_superseded ON public.sops (superseded_by);

-- Helper function: get the current (non-superseded) version of an SOP lineage
CREATE OR REPLACE FUNCTION public.current_sop_version(p_parent_id uuid)
RETURNS uuid AS $$
  SELECT id FROM public.sops
  WHERE (parent_sop_id = p_parent_id OR id = p_parent_id)
    AND superseded_by IS NULL
    AND status = 'published'
  ORDER BY version DESC
  LIMIT 1;
$$ LANGUAGE sql STABLE;
```

**2. Create `supabase/migrations/00009_worker_notifications.sql`:**
```sql
CREATE TABLE public.worker_notifications (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  organisation_id  uuid NOT NULL REFERENCES public.organisations(id) ON DELETE CASCADE,
  user_id          uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  sop_id           uuid NOT NULL REFERENCES public.sops(id) ON DELETE CASCADE,
  type             text NOT NULL DEFAULT 'sop_updated',
  read             boolean NOT NULL DEFAULT false,
  created_at       timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX idx_notifications_user ON public.worker_notifications (user_id, read);
CREATE INDEX idx_notifications_org ON public.worker_notifications (organisation_id);

ALTER TABLE public.worker_notifications ENABLE ROW LEVEL SECURITY;

-- Workers see their own notifications
CREATE POLICY "users_see_own_notifications"
  ON public.worker_notifications FOR SELECT TO authenticated
  USING (user_id = auth.uid() AND organisation_id = public.current_organisation_id());

-- Workers can mark their own notifications as read
CREATE POLICY "users_can_update_own_notifications"
  ON public.worker_notifications FOR UPDATE TO authenticated
  USING (user_id = auth.uid() AND organisation_id = public.current_organisation_id())
  WITH CHECK (user_id = auth.uid() AND organisation_id = public.current_organisation_id());

-- Admins can insert notifications (when publishing a new version)
CREATE POLICY "admins_can_insert_notifications"
  ON public.worker_notifications FOR INSERT TO authenticated
  WITH CHECK (
    organisation_id = public.current_organisation_id()
    AND public.current_user_role() IN ('admin', 'safety_manager')
  );
```

**3. Create `src/actions/versioning.ts`:**
- `'use server'` directive
- Import `createClient` from `@/lib/supabase/server`, `z` from `zod`

- Export `async function uploadNewVersion(oldSopId: string, formData: FormData)`:
  - Get supabase client, verify admin/safety_manager role
  - Fetch old SOP record to get `version`, `parent_sop_id`, `organisation_id`
  - Compute `newParentId = oldSop.parent_sop_id ?? oldSop.id` (first version's id becomes parent for all)
  - Compute `newVersion = oldSop.version + 1`
  - Create new SOP record using existing `createUploadSession` pattern but with:
    - `version: newVersion`
    - `parent_sop_id: newParentId`
  - Update old SOP: `supabase.from('sops').update({ superseded_by: newSopId }).eq('id', oldSopId)`
  - Return `{ success: true, newSopId, uploadUrl, token, path }` (same shape as UploadSession)

- Export `async function notifyAssignedWorkers(sopId: string)`:
  - Get supabase client, verify admin role
  - Get old SOP's `parent_sop_id` to find assignments for the SOP lineage
  - Find all assignments: query `sop_assignments` where `sop_id` matches the old SOP id
  - For role assignments: query `user_profiles` matching the role to get user IDs
  - For individual assignments: use `user_id` directly
  - Deduplicate user IDs
  - Bulk insert into `worker_notifications`: one record per user with `sop_id` = new SOP, `type` = 'sop_updated'
  - Also update assignments to point to new SOP: `supabase.from('sop_assignments').update({ sop_id: newSopId }).eq('sop_id', oldSopId)`
  - Return `{ success: true, notified: userIds.length }`

- Export `async function getVersionHistory(sopId: string)`:
  - Get supabase client
  - Get the SOP to find `parent_sop_id`
  - Query all SOPs where `parent_sop_id = parentId OR id = parentId`, ordered by `version DESC`
  - Return `{ success: true, versions: data }` with version number, status, uploaded_by, created_at, superseded_by

- Export `async function markNotificationRead(notificationId: string)`:
  - Get supabase client
  - Update `worker_notifications` set `read = true` where `id = notificationId`
  - Return `{ success: true }`
